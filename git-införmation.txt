#!/bin/bash
# git_new_project.sh
# Script för att skapa ett nytt projekt och koppla det till GitHub med SSH

# 1. Skapa ny mapp (byt ut "my-project" till projektets namn)
mkdir my-project
cd my-project

# 2. Initiera ett nytt git-repo
git init

# 3. Lägg till alla filer i projektet
git add .

# 4. Första commit
git commit -m "Initial commit"

# 5. Skapa repo på GitHub manuellt via webben och kopiera SSH-URL
# Exempel: git@github.com:USERNAME/my-project.git

# 6. Koppla repo till GitHub (byt ut URL mot din)
git remote add origin git@github.com:USERNAME/my-project.git

# 7. Byt branch till main (bra standard)
git branch -M main

# 8. Pusha till GitHub
git push -u origin main









# Spara som git_new_project.sh.
#Gör den körbar:

chmod +x git_new_project.sh
./git_new_project.sh






#!/bin/bash
# git_commands_cheat.sh
# Git kommando-cheatsheet

# Visa status
git status

# Lägg till fil(er) till staging area
git add <filnamn>    # Ex: git add file.py
git add .            # Lägg till alla ändringar

# Commit ändringar
git commit -m "Beskrivning av ändringar"

# Byt branch
git checkout <branch-namn>
git checkout -b <ny-branch-namn>   # Skapar + byter branch

# Lista brancher
git branch

# Hämta ändringar från remote
git pull origin <branch-namn>

# Skicka ändringar till remote
git push origin <branch-namn>

# Koppla remote repo
git remote add origin git@github.com:USERNAME/REPO.git

# Ändra remote URL
git remote set-url origin git@github.com:USERNAME/REPO.git

# Visa remote
git remote -v

# Klona ett repo
git clone git@github.com:USERNAME/REPO.git

# Ta bort branch
git branch -d <branch-namn>

# Visa commit-historik
git log --oneline --graph --all
